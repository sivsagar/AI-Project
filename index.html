Propositional Logic: Resolution (JavaScript)
function resolve(premises, conclusion) {
  let clauses = premises.map(p => toCNF(p));
  clauses.push(toCNF(`~(${conclusion})`));

  let newClauses = [];
  while (true) {
    for (let i = 0; i < clauses.length; i++) {
      for (let j = i + 1; j < clauses.length; j++) {
        
        // Search for P and ~P
        let resolvents = resolvePair(clauses[i], clauses[j]);
        
        for (let res of resolvents) {
          if (res.length === 0) {
            return true; // Contradiction found!
          }
          newClauses.push(res);
        }
      }
    }
    if (isSubset(newClauses, clauses)) return false;
    clauses.push(...newClauses);
  }
}
First-Order Logic: Forward Chaining (JS)
function folInference(premises, query) {
  let facts = new Set();
  let rules = [];

  // 1. Skolemization (Existential Instantiation)
  for (let p of premises) {
    if (p.startsWith('∃')) {
      const c = `C${counter++}`;
      facts.add(p.body.replace(p.var, c));
    } else {
      rules.push(p);
    }
  }
  
  // 2. Forward Chaining (Modus Ponens)
  while (changed) {
    for (let rule of rules) {
      if (unify(rule.antecedent, facts)) {
        facts.add(rule.consequent);
      }
    }
  }

  // 3. Existential Generalization
  return matchQuery(query, facts);
}
